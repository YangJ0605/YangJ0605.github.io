function Human(name){
     this.name = name
 }
 Human.prototype.run = function(){
     console.log("我叫"+this.name+"，我在跑")
     return undefined
 }
 function Man(name){
     Human.call(this, name) //借用构造函数实现继承，只能继承父类子自身的属性，无法继承原型中的属性和方法
     this.gender = '男'
 }

 var f = function(){}  // 构造一个空的构造函数
 f.prototype = Human.prototype  //使这个空的构造函数的原型指向父类的原型
 Man.prototype = new f()        // 使子类的原型指向这个空的构造函数的实例

//上面三句话相当于
// Man.prototype.__proto__=Human.prototype

 Man.prototype.fight = function(){
     console.log('糊你熊脸')
 }

 class Human{
     constructor(name){
         this.name = name  //写在constructor里的是实例对象的属性
     }
     run(){  //写在外面与constructor平级的是共有的属性。这种方法不能直接在原型上添加某个方法；只能写一个函数，然后返回
            // 相当于Human.prototype.run
         console.log("我叫"+this.name+"，我在跑")  
         return undefined
     }
     get(){
         return '你想要的属性'
     }
 }
 class Man extends Human{  // Man.prototype.__proto__=Human.prototype
     constructor(name){
         super(name)  //Human.call(this, name)
         this.gender = '男'
     }
     fight(){
         console.log('糊你熊脸')
     }
 }

ES6规定Class类没有静态属性,只有静态方法：static

所谓静态，不需要实例化对象，直接调用

class Foo {
      static classMethod() {
            return 'lity';
       }
 }
 var f= new Foo()
 console.log(Foo.classMethod()) // 'hello'
 console.log(f.classMethod())  //报错

函数前面添加一个static关键字，表明这是一个静态方法，不会被实例继承，只能通过类来调用
